import React from 'react';
import PropTypes from 'prop-types'

const Header = (data) => {
    return (
        <div className='header'>
            <Name data={data.data}/>
            <Message />
        </div>

};

Header.propTypes = {
    data: PropTypes.object.isRequired
}

export default Header;
